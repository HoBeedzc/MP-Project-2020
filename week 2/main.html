def every_cut(cut_list,language = 'Chinese',to_file = ''):
    '''
    '''
    res = []
    if language == 'Chinese':
        for i in cut_list:
            temp = []
            for j in jb.cut(i, cut_all=True):
                if j not in ['',' ']:
                    temp.append(j)
            res.append(temp)
    elif language == 'English':
        for i in cut_list:
            res.append(i.split())
    else:
        print('Try again!')
        return -1
    if to_file:
        with open(to_file,'w',encoding='unicode-escape') as f:
            for i in res:
                f.write(','.join(i))
    return res

# IDF = log( / +1)
# TF-IDF = TF * IDF
def statistical_frequency_IDF(data_df,data_ap):
    '''
    '''
    res_dict = {}
    nums = len(data_ap) # 
    for row in data_df.itertuples():
        name = row[0]
        temp = 0
        for i in data_ap:
            if name in i:
                temp += 1
        res_dict[name] = temp
    for i in res_dict.keys():
        res_dict[i] = math.log(nums/(res_dict[i] + 1))
    data_df['IDF'] = res_dict.values()
    return data_df
    

def find_feature_sets_by_TFIDF(df):
    '''
    '''
    res = []
    for row in df.itertuples():
        res.append(row[1]*row[2])
    df['TF-IDF'] = res
    return df

# tf-idf
def doc2tfidf_matrix(vocab, words_list,to_file = ''):
    '''
    '''
    # MV0MV
    V = len(vocab)
    M = len(words_list)
    onehot = np.zeros((M,V)) # 
    tf = np.zeros((M,V))
    
    for i,doc in enumerate(words_list):
        for word in doc:
            if word in vocab:
                pos = vocab.index(word)
                onehot[i][pos] = 1
                tf[i][pos] += 1 # tf,
    
    row_sum = tf.sum(axis=1) # 
    # TF(t,d)
    tf = tf/row_sum[:, np.newaxis] #tf[:,np.newaxis]
    # DF(t,D)IDF
    df = onehot.sum(axis=0) # 
    idf = list(map(lambda x:math.log10((M+1)/(x+1)),df))
    # TFIDF
    tfidf = tf*np.array(idf)
    tfidf = pd.DataFrame(tfidf,columns=vocab)
    if to_file:
        tfidf.to_csv(to_file)
    return tfidf

def calculate_distance(vec_df,fuc = '',to_file = ''):
    '''
    '''
    LEN = vec_df.shape[0]
    res = [[0 for _ in range(LEN)] for __ in range(LEN)]
    for i in range(LEN):
        for j in range(i):
            res[i][j] = math.sqrt(sum((vec_df.loc[i] - vec_df.loc[j])**2))
    res = pd.DataFrame(res)
    if to_file:
        res.to_csv(to_file)
    return res

def find_heart(vec_df,odata):
    '''
    '''
    heart = vec_df.mean(axis = 0)
    min_index = -1
    min_num = 99999
    for i in range(vec_df.shape[0]):
        temp = math.sqrt(sum((vec_df.loc[i] - heart)**2))
        if temp <= min_num:
            min_num = temp
            min_index = i
    return odata[min_index]
